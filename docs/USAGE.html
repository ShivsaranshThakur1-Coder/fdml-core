FDML — Usage

./bin/fdml validate <path> [--json] [--json-out file]
./bin/fdml validate-sch <path> [--json]
./bin/fdml validate-all <path> [--json]
./bin/fdml render <fdml-file> [--out out.html]
./bin/fdml export-pdf <fdml-file> [--out out.pdf]
./bin/fdml index <path> --out out/index.json
./bin/fdml lint <path> [--json] [--strict]
./bin/fdml init <output-file> [--title T] [--dance D] [--meter M/N] [--tempo BPM] [--figure-id f-...] [--figure-name NAME] [--formation FORM]
./bin/fdml doctor <path> [--json] [--strict]
